# Stage 1: Build the Angular application
# This section uses an official Node.js image to install dependencies and build your Angular project.
FROM node:18 AS build
WORKDIR /app
COPY package*.json ./
RUN npm install
COPY . .
RUN npm run build

# Stage 2: Serve the application with Nginx
# This section takes the built files from the first stage and puts them into a lightweight Nginx web server.
FROM nginx:alpine
COPY --from=build /app/dist/healthcare-frontend/browser /usr/share/nginx/html
EXPOSE 80
